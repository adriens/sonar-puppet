Noncompliant Code Example

service { 'sshd':
  ensure => running,
  enable => true,
}
<~                                                   # Noncompliant
file { '/etc/ssh/sshd_config':
  ensure => file,
  mode   => '0600',
  source => 'puppet:///modules/sshd/sshd_config',
}
<-                                                   # Noncompliant
package { 'openssh-server':
  ensure => present,
}

Compliant Solution

package { 'openssh-server':
  ensure => present,
}
->
file { '/etc/ssh/sshd_config':
  ensure => file,
  mode   => '0600',
  source => 'puppet:///modules/sshd/sshd_config',
}
~>
service { 'sshd':
  ensure => running,
  enable => true,
}
